options API Composition API

        
          //options API=>vue2
        export default {
          data() {
            return {
              message: 'Hello, Vue 2!',
            };
          },
          methods: {
            reverseMessage() {
              this.message = this.message.split('').reverse().join('');
            },
          },
        };

        //composition API=>vue3
        import { ref } from 'vue';
        export default {
          setup() {
            // 创建响应式数据
            const message = ref('Hello, Vue 3!');
        
            // 定义一个函数来操作数据
            function reverseMessage() {
              message.value = message.value.split('').reverse().join('');
            }
        
            return {
              message,
              reverseMessage,
            };
          },
        };

        //composition API setup=>語法糖
          <script setup>
          import {computed,reactive,ref,watch,watchEffect} from 'vue'
          //obj 方式
          const props = defineProps({
            data:{
              type: String,
              default: '',
              requires: true,
            }
          })
          const msgCount = computed(()=>{
            return props.data.length
          })
          </script>

silde-effect and pureFunc

        
          const a = [1,2,3]
        
        //silde-effect 副作用/髒的/污染到變數
        const test = (num)=>{
          a.push(num)
        }
        test(4)
        consoel.log(a) [1,2,3,4]

        //pureFunc 乾淨的/純淨
        const pureFunc = (num)=>{
          const newA = Array.from(a)
          newA.push(num)
          return newA
        }
        const newA = pureFunc(4)
        consoel.log(newA) [1,2,3,4]
